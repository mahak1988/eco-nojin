        from api.services.early_warning_engine import ews_engine

    except Exception as e:
        print(f"รขฺย รฏยธฺ EWS skipped: {e}")

    print("Ready on http://127.0.0.1:8000")
    yield
    print("Shutting down...")

@router.post("/otp/request", response_model=schemas.OtpRequestResponse)
async def request_otp(req: schemas.OtpRequest):
    # ุทยงุบลุธโฆุธยพุธหุทยฑุทฺพ ุทยฏุทยงุทยฎุธโุบล ุทยจุทยฑุทยงุบล ุทยฌุธโุธหุนยฏุบลุทยฑุบล ุทยงุทยฒ Circular Import (ุทยงุนยฏุทยฑ ุธโ ุบลุทยงุทยฒ ุทยงุทยณุทฺพ)
    from api.services.sms import send_otp_sms

    code = generate_otp(req.phone)
    sent = await send_otp_sms(req.phone, code)

    if not sent and not settings.OTP_DEV_MODE:
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="ุทยณุทยฑุธหุบลุทยณ ุธยพุบลุทยงุธโฆุนยฉ ุทยฏุทยฑ ุทยฏุทยณุทฺพุทยฑุทยณ ุธโ ุบลุทยณุทฺพ. ุธยพุบลุนยฉุทยฑุทยจุธโ ุทยฏุบล Kavenegar ุบลุทยง Twilio ุทยฑุทยง ุทยจุทยฑุทยฑุทยณุบล ุนยฉุธโ ุบลุทยฏ.",
        )

    payload = {"sent": True, "phone": req.phone, "expires_in": 300}
    if settings.DEBUG or settings.OTP_DEV_MODE:
        payload["dev_code"] = code  # ุธูพุธโุทยท ุทยฏุทยฑ ุธโฆุทยญุบลุทยท ุทฺพุธหุทยณุทยนุธโก ุทยงุทยฑุทยณุทยงุธโ ุทยดุธหุทยฏ

    return payload


@router.post("/otp/verify", response_model=schemas.TokenResponse)
async def verify_otp_login(req: schemas.OtpVerify, db: AsyncSession = Depends(get_db)):
    if not verify_otp(req.phone, req.code):
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST, detail="ุนยฉุทยฏ ุธหุทยงุทยฑุทยฏ ุทยดุทยฏุธโก ุธโ ุทยงุธโฆุทยนุทฺพุทยจุทยฑ ุบลุทยง ุธโฆุธโ ุธโุทยถุบล ุทยดุทยฏุธโก ุทยงุทยณุทฺพ"
        )

    # Upsert ุนยฉุทยงุทยฑุทยจุทยฑ (ุทยงุบลุทยฌุทยงุทยฏ ุบลุทยง ุทยจุธโกรขโฌลุทยฑุธหุทยฒุทยฑุทยณุทยงุธโ ุบล)
    login_req = schemas.LoginRequest(fid=req.fid, phone=req.phone, name=req.name)
    user = await crud.upsert_user(db, login_req)

    # ุทยณุทยงุทยฎุทฺพ ุทฺพุธหุนยฉุธโ  ุทยจุทยง farmer_id ุทยจุธโก ุทยนุธโ ุธหุทยงุธโ  subject
    token = create_access_token(subject=user.farmer_id)

    return schemas.TokenResponse(access_token=token, farmer_id=user.farmer_id)


@router.post("/login", response_model=schemas.TokenResponse)
async def login(req: schemas.LoginRequest, db: AsyncSession = Depends(get_db)):
    """
    รขฺย รฏยธฺ ุธโกุทยดุทยฏุทยงุทยฑ ุทยงุธโฆุธโ ุบลุทฺพุบล: ุทยงุบลุธโ  ุทยงุธโ ุทยฏุธยพุธหุบลุธโ ุทฺพ ุทยจุทยฏุธหุธโ  ุทยจุทยฑุทยฑุทยณุบล ุทยฑุธโฆุทยฒ ุทยนุทยจุธหุทยฑ ุบลุทยง OTPุทล ุนยฉุทยงุทยฑุทยจุทยฑ ุทยฑุทยง upsert ุธโฆุบลรขโฌลุนยฉุธโ ุทยฏ.
    ุทยงุนยฏุทยฑ ุทยงุบลุธโ  ุทยฑุธูพุทฺพุทยงุทยฑ ุทยนุธโฆุทยฏุบล ุธโ ุบลุทยณุทฺพุทล ุทยจุทยงุบลุทยฏ ุธูพุบลุธโุทยฏ password ุทยจุธโก Schema ุทยงุทยถุทยงุธูพุธโก ุทยดุทยฏุธโก ุธห ุทยฏุทยฑ CRUD ุทยจุทยฑุทยฑุทยณุบล ุทยดุธหุทยฏ.
    """
    user = await crud.upsert_user(db, req)
    token = create_access_token(subject=user.farmer_id)
    return schemas.TokenResponse(access_token=token, farmer_id=user.farmer_id)


@router.get("/profile", response_model=schemas.ProfileResponse)
async def get_profile(
    farmer_id: str = Depends(get_current_user_id),
    db: AsyncSession = Depends(get_db),
):
    """ุทยฏุทยฑุบลุทยงุธูพุทฺพ ุธยพุทยฑุธหุธูพุทยงุบลุธโ ุนยฉุทยงุทยฑุทยจุทยฑ ุทยจุทยง ุทยงุทยณุทฺพุธูพุทยงุทยฏุธโก ุทยงุทยฒ farmer_id ุทยงุทยณุทฺพุทยฎุทยฑุทยงุทยฌ ุทยดุทยฏุธโก ุทยงุทยฒ ุทฺพุธหุนยฉุธโ """
    user = await crud.get_user_by_farmer_id(db, farmer_id)
    if not user:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND, detail="ุนยฉุทยงุทยฑุทยจุทยฑ ุทยจุทยง ุทยงุบลุธโ  ุทยดุธโ ุทยงุทยณุธโก ุบลุทยงุธูพุทฺพ ุธโ ุทยดุทยฏ"
        )

    return schemas.ProfileResponse(
        fid=user.farmer_id,
        name=user.name or user.farmer_id,
        phone=user.phone,
        registered_at=user.created_at,
        wallet_address=user.wallet_address,
    )

@router.delete("/{farmer_id}", status_code=status.HTTP_204_NO_CONTENT, response_model=IDResponse)
async def delete_farmer(
    farmer_id: int,
    db: AsyncSession = Depends(get_db),
    _user: str = Depends(require_write_auth),
):
    if not await crud.delete_farmer(db, farmer_id):
        raise HTTPException(status_code=404, detail=f"Farmer {farmer_id} not found")


@router.get("/{farmer_id}/activities", response_model=SuccessResponse)
async def get_farmer_activities(farmer_id: int, db: AsyncSession = Depends(get_db)):
    farmer = await crud.get_farmer(db, farmer_id)
    if not farmer:
        raise HTTPException(status_code=404, detail=f"Farmer {farmer_id} not found")
    return {"farmer_id": farmer_id, "activities": []}
